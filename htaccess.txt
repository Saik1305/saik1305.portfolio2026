RewriteEngine On

# Redirect /index.html to the root /
RewriteCond %{THE_REQUEST} \s/+index\.html[\s?] [NC]
RewriteRule ^ / [R=301,L]

# Redirect any other .html request to the clean URL with trailing slash
# e.g. /about.html -> /about/  |  /projects/name.html -> /projects/name/
RewriteCond %{THE_REQUEST} \s/+(.*?)\.html[\s?] [NC]
RewriteRule ^ /%1/ [R=301,L]

# Internally rewrite a top-level clean URL to its .html file
# e.g. /about/ -> about.html
RewriteCond %{REQUEST_FILENAME} !-d
RewriteCond %{REQUEST_FILENAME} !-f
RewriteRule ^([^/]+)/$ $1.html [L]

# Internally rewrite a two-level clean URL to its .html file
# e.g. /projects/my-project/ -> projects/my-project.html
RewriteCond %{REQUEST_FILENAME} !-d
RewriteCond %{REQUEST_FILENAME} !-f
RewriteRule ^([^/]+)/([^/]+)/$ $1/$2.html [L]
